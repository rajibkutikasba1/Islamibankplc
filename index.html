// ব্রাউজার চেক if (!window.PublicKeyCredential) { alert("🚨 এই ব্রাউজার বা ডিভাইসটি ফিঙ্গারপ্রিন্ট লগইন সাপোর্ট করে না। দয়া করে Chrome বা Edge ব্যবহার করুন।"); } // টেক্সটকে ArrayBuffer এ রূপান্তর করার সহজ ফাংশন function strToBuffer(str) { return new TextEncoder().encode(str); } // ১. রেজিস্ট্রেশন document.getElementById('registerBtn').addEventListener('click', async () => { const username = document.getElementById('username').value; if (!username) return alert("অনুগ্রহ করে নাম বা আইডি লিখুন"); // সিকিউর কানেকশন চেক if (location.protocol !== 'https:' && location.hostname !== 'localhost') { return alert("🚨 নিরাপত্তা জনিত কারণে এই কোডটি রান করতে আপনার সাইটে HTTPS বা লাইভ সার্ভার লিংক লাগবে।"); } const options = { challenge: strToBuffer("secure_random_challenge_string_12345"), rp: { name: "Biometric System" }, user: { id: strToBuffer("USER_" + Date.now()), name: username, displayName: username }, pubKeyCredParams: [{ alg: -7, type: "public-key" }], // ES256 authenticatorSelection: { authenticatorAttachment: "platform" }, timeout: 60000 }; try { const credential = await navigator.credentials.create({ publicKey: options }); const credentialId = btoa(String.fromCharCode(...new Uint8Array(credential.rawId))); // সার্ভারে ডাটা পাঠানো const response = await fetch(`${BASE_URL}/api/register`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username, credentialId }) }); const resData = await response.json(); if(resData.success) alert("✅ আপনার ফিঙ্গারপ্রিন্ট সার্ভারে সেভ হয়েছে!"); } catch (err) { console.error(err); alert("❌ রেজিস্ট্রেশন ব্যর্থ! আপনি কি পপ-আপ বাতিল করেছেন বা ডিভাইসে লক সেট করা নেই?"); } }); // ২. স্ক্যান ও সার্চ document.getElementById('scanBtn').addEventListener('click', async () => { const options = { challenge: strToBuffer("secure_random_challenge_string_12345"), timeout: 60000 }; try { const assertion = await navigator.credentials.get({ publicKey: options }); const credentialId = btoa(String.fromCharCode(...new Uint8Array(assertion.rawId))); // সার্ভার থেকে খোঁজা const response = await fetch(`${BASE_URL}/api/search`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ credentialId }) }); const resData = await response.json(); if(resData.found) { document.getElementById('result').style.display = 'block'; document.getElementById('userInfo').innerHTML = ` নাম: ${resData.user.name}
পদবী: ${resData.user.role}
আইডি নং: ${resData.user.idNo}
স্ট্যাটাস: ডাটা লাইভ সার্ভার থেকে লোড হয়েছে। `; } } catch (err) { console.error(err); alert("❌ ভেরিফিকেশন ব্যর্থ হয়েছে বা ফিঙ্গারপ্রিন্ট মেলেনি।"); } });
